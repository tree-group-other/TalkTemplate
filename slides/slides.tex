  \begin{columns}[T]

      \begin{tikzpicture}

        \node[fill=eqcol1,rounded corners,anchor=base] (m1) {
          $\displaystyle \dot{F}[\{ \bm{v}_i\}]$
        };

        \node[below=0.1 of m1] (l1) {free energy};
    
        \node[fill=eqcol2!80,rounded corners,anchor=base, right=0.4 of m1] (m2) {

        };

        \node[below=0.1 of m2] (l2) {dissipation};

        \node[fill=eqcol3,rounded corners,anchor=base, right=0.4 of m2] (m3) {
          $\displaystyle \lambda G[\{\bm{v}_{i}\}]$\vphantom{$\dot{F}$}
        };

        \node[below=0.1 of m3] (l3) {constraints};

        \node[below=0.2 of l2, align=center, text width=\textwidth, 
              draw=darkblue, rounded corners, thick] (numerical) {
          Solve on GPUs using pseudospectral method
        };

      \end{tikzpicture}


    \column{0.55\textwidth}

      \centering

      \tikz{\node[draw=black, rounded corners, align=center]{ 
         Convection-Diffusion Equation}; 
      }

      \vspace{-18pt}
      \begin{gather*}
      \frac{\partial \phi_{i}}{\partial t} + \bm{v} \cdot \nabla \phi_{i}  = \
             \nabla \cdot \left ( \sum_{j} M_{ij}(\phi_{i}) \nabla \mu_{j} \right ) \\
        \mu_{i} = \frac{\delta F[\phi_{i}]}{\delta \phi_{i}}, \,
          F = \int d\bm{r} \left [ f(\phi_{i}) + 
              \frac{1}{2} \sum_{i} \kappa_{i} \left | \nabla \phi_{i} \right | ^{2} \right ]
      \end{gather*} \par

      \tikz{\node[draw=black, rounded corners, align=center]{ 
        Incompressible Stokes Equation}; 
      }
      \begin{gather*}
        0 = -\nabla p + \nabla \cdot \left [ \eta(\phi_{i}) (\nabla \bm{v} + \nabla \bm{v}^{T}) \right ] \
            - \nabla \cdot \bm{\Pi} \\[3pt]
        0 = \nabla \cdot \bm{v}
      \end{gather*} \par

      \makebox[\textwidth][c]{\small Tree et al.\ Soft Matter.\ 13, 3013--3030 (2017)}

    \end{columns}
